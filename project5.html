Proof of code


            

import spacy

# Load the spaCy model
nlp = spacy.load('en_core_web_sm')

def tokenize_and_lemmatize(text):
    """Tokenize, lowercase, and lemmatize text, removing stop words and punctuation."""
    doc = nlp(text.lower())
    words = [token.lemma_ for token in doc if not token.is_stop and not token.is_punct and len(token.text) > 2]
    return words

def word_frequency(words):
    """Count the frequency of each word."""
    freq = {}
    for word in words:
        freq[word] = freq.get(word, 0) + 1
    return freq

# Example usage
text = "This is a sample text for analysis. It demonstrates text processing with spaCy."
words = tokenize_and_lemmatize(text)
freq = word_frequency(words)

print("Processed words:", words)
print("Word frequencies:", freq)
